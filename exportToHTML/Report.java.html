import javax.swing.*;
import java.io.File;
import java.io.FileWriter;
import java.io.IOException;
import org.w3c.dom.*;
import org.xml.sax.SAXException;

import javax.xml.parsers.*;
import java.io.*;

public class Report {
    private File file;

    public File getFile(){
        return file;
    }

    public File openFile() {
        JFileChooser fileopen = new JFileChooser();
        fileopen.showDialog(null, "Select the file with the .tab extension");
        int ret = fileopen.showDialog(null, "Select the file with the .tab extension");
        if (ret == JFileChooser.APPROVE_OPTION) {
            file = fileopen.getSelectedFile();
        }
        return file;

    }

    public String getFileName(File file) {
        String fileName = getFile().getName();
        return fileName;
    }

    public void doReport(String fileName) {
        try (FileWriter writer = new FileWriter(fileName, false)) {
            // запись всей строки
            String text = "Hello Gold!";
            writer.write(text);
            // запись по символам
            writer.append('\n');
            writer.append('E');

            writer.flush();
        } catch (IOException ex) {

            System.out.println(ex.getMessage());
        }
    }




}
